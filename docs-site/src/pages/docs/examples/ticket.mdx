# Event Ticket Generator

Create custom event tickets with QR codes, barcodes, and professional designs.

## Complete Example

```tsx
import { usePDF } from 'react-to-pdf';
import QRCode from 'qrcode.react'; // npm install qrcode.react

interface TicketProps {
  eventName: string;
  eventDate: string;
  venue: string;
  ticketNumber: string;
  attendeeName: string;
  seatInfo: string;
}

function EventTicket({
  eventName,
  eventDate,
  venue,
  ticketNumber,
  attendeeName,
  seatInfo
}: TicketProps) {
  const { targetRef, toPDF } = usePDF({
    filename: `ticket-${ticketNumber}.pdf`,
    page: { format: [600, 250], margin: 0 }, // Custom size: 600x250px
    canvas: { scale: 3 }
  });

  return (
    <div className="ticket-container">
      <button onClick={() => toPDF()} className="download-btn">
        Download Ticket
      </button>

      <div ref={targetRef} className="ticket">
        <div className="ticket-main">
          <div className="ticket-header">
            <h2>{eventName}</h2>
            <p className="event-date">{eventDate}</p>
          </div>

          <div className="ticket-details">
            <div className="detail-row">
              <span className="label">Venue:</span>
              <span className="value">{venue}</span>
            </div>
            <div className="detail-row">
              <span className="label">Attendee:</span>
              <span className="value">{attendeeName}</span>
            </div>
            <div className="detail-row">
              <span className="label">Seat:</span>
              <span className="value">{seatInfo}</span>
            </div>
            <div className="detail-row">
              <span className="label">Ticket #:</span>
              <span className="value">{ticketNumber}</span>
            </div>
          </div>
        </div>

        <div className="ticket-stub">
          <QRCode
            value={`TICKET:${ticketNumber}`}
            size={80}
            level="H"
          />
          <p className="ticket-number">{ticketNumber}</p>
        </div>
      </div>
    </div>
  );
}

export default EventTicket;
```
