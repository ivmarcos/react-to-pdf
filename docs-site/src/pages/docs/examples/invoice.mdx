# Invoice Generator

Generate professional, printable invoices with React to PDF. This example shows how to create a complete invoice with company branding, line items, and calculations.

## Complete Example

```tsx
import { usePDF } from 'react-to-pdf';
import './Invoice.css'; // Or use your preferred styling method

interface InvoiceItem {
  description: string;
  quantity: number;
  rate: number;
  amount: number;
}

interface InvoiceData {
  invoiceNumber: string;
  date: string;
  dueDate: string;
  from: {
    name: string;
    address: string;
    city: string;
    email: string;
  };
  to: {
    name: string;
    address: string;
    city: string;
    email: string;
  };
  items: InvoiceItem[];
  subtotal: number;
  tax: number;
  total: number;
}

function InvoiceGenerator() {
  const { targetRef, toPDF } = usePDF({
    filename: 'invoice.pdf',
    page: {
      format: 'a4',
      margin: 10
    }
  });

  const invoiceData: InvoiceData = {
    invoiceNumber: 'INV-2024-001',
    date: '2024-01-15',
    dueDate: '2024-02-15',
    from: {
      name: 'Your Company Inc.',
      address: '123 Business St',
      city: 'San Francisco, CA 94102',
      email: 'billing@yourcompany.com'
    },
    to: {
      name: 'Acme Corporation',
      address: '456 Client Ave',
      city: 'New York, NY 10001',
      email: 'accounts@acmecorp.com'
    },
    items: [
      {
        description: 'Web Development Services',
        quantity: 40,
        rate: 150,
        amount: 6000
      },
      {
        description: 'UI/UX Design',
        quantity: 20,
        rate: 120,
        amount: 2400
      },
      {
        description: 'Consulting',
        quantity: 10,
        rate: 200,
        amount: 2000
      }
    ],
    subtotal: 10400,
    tax: 1040,
    total: 11440
  };

  return (
    <div className="invoice-container">
      <div className="invoice-actions">
        <button onClick={() => toPDF()} className="btn-download">
          Download Invoice
        </button>
      </div>

      <div ref={targetRef} className="invoice">
        {/* Header */}
        <div className="invoice-header">
          <div className="company-info">
            <h1>{invoiceData.from.name}</h1>
            <p>{invoiceData.from.address}</p>
            <p>{invoiceData.from.city}</p>
            <p>{invoiceData.from.email}</p>
          </div>
          <div className="invoice-meta">
            <h2>INVOICE</h2>
            <p><strong>Invoice #:</strong> {invoiceData.invoiceNumber}</p>
            <p><strong>Date:</strong> {invoiceData.date}</p>
            <p><strong>Due Date:</strong> {invoiceData.dueDate}</p>
          </div>
        </div>

        {/* Bill To */}
        <div className="invoice-to">
          <h3>Bill To:</h3>
          <p><strong>{invoiceData.to.name}</strong></p>
          <p>{invoiceData.to.address}</p>
          <p>{invoiceData.to.city}</p>
          <p>{invoiceData.to.email}</p>
        </div>

        {/* Items Table */}
        <table className="invoice-table">
          <thead>
            <tr>
              <th>Description</th>
              <th>Quantity</th>
              <th>Rate</th>
              <th>Amount</th>
            </tr>
          </thead>
          <tbody>
            {invoiceData.items.map((item, index) => (
              <tr key={index}>
                <td>{item.description}</td>
                <td>{item.quantity}</td>
                <td>${item.rate.toFixed(2)}</td>
                <td>${item.amount.toFixed(2)}</td>
              </tr>
            ))}
          </tbody>
        </table>

        {/* Totals */}
        <div className="invoice-totals">
          <div className="total-row">
            <span>Subtotal:</span>
            <span>${invoiceData.subtotal.toFixed(2)}</span>
          </div>
          <div className="total-row">
            <span>Tax (10%):</span>
            <span>${invoiceData.tax.toFixed(2)}</span>
          </div>
          <div className="total-row total">
            <span><strong>Total:</strong></span>
            <span><strong>${invoiceData.total.toFixed(2)}</strong></span>
          </div>
        </div>

        {/* Footer */}
        <div className="invoice-footer">
          <p>Payment is due within 30 days of the invoice date.</p>
          <p>Thank you for your business!</p>
        </div>
      </div>
    </div>
  );
}

export default InvoiceGenerator;
```

## Best Practices

1. **Use consistent branding** - Include your logo, colors, and fonts
2. **Clear structure** - Organize information logically (from, to, items, total)
3. **Professional typography** - Use readable fonts and appropriate sizing
4. **Table formatting** - Make line items easy to scan
5. **Include all details** - Invoice number, dates, payment terms
6. **Test print quality** - Use higher scale for better quality: `canvas: { scale: 3 }`

## Dynamic Data

In a real application, you'd fetch invoice data from an API:

```tsx
import { useState, useEffect } from 'react';
import { usePDF } from 'react-to-pdf';

function DynamicInvoice({ invoiceId }: { invoiceId: string }) {
  const [invoiceData, setInvoiceData] = useState<InvoiceData | null>(null);
  const { targetRef, toPDF } = usePDF({
    filename: `invoice-${invoiceId}.pdf`
  });

  useEffect(() => {
    fetch(`/api/invoices/${invoiceId}`)
      .then(res => res.json())
      .then(data => setInvoiceData(data));
  }, [invoiceId]);

  if (!invoiceData) {
    return <div>Loading invoice...</div>;
  }

  return (
    // ... invoice markup ...
  );
}
```
